\begin{questions}
    \question Describe geometrically the sets of points $z$ in the complex plane defined by the following relations:
    \begin{parts}
        \part $|z - z_1| = |z - z_2|$ where $z_1, z_2 \in\C$. \\
        A line going passing through the intersection of the circles $|z - z_1| = r$ and $|z - z_2| = r$. 

        \part $1 / z = \overline{z}$. \\
        

        \part $\re{z} = 3$. \\
        All complex numbers of the form $z = 3 + bi$, where $b\in\R$. Geometrically, this is a vertical line at 3.

        \part $\re{z} > c$ where $c\in\R$. \\
        All complex numbers of the form $z = c + bi$, where $b\in\R$. Geometrically, everything to the right of the 
        vertical line at $c$.

        \part $\re{az + b} > 0$ where $a,b\in\C$.

        \part $|z| = \re{z} + 1$.

        \part $\im{z} = c$ with $c\in\R$. \\
        All complex numbers of the form $z = a + ci$, where $a\in\R$. Geometrically, this is a horizontal line at $ci$.
    \end{parts}
\end{questions}
