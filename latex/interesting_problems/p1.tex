\begin{questions}
    \question If $v, u\in\R^n$ for $n < \infty$ with
    $$v_i = \begin{cases}
        1, & \text{ with probability 1/2}
        0, & \text{ with probability 1/2}
    \end{cases} \text{ and}$$
    $$u_i = \begin{cases}
        1, & \text{ with probability 1/4}
        0, & \text{ with probability 3/4}
    \end{cases},$$
    what is the expected value of $v\cdot u$ (with proof)?
\end{questions}
